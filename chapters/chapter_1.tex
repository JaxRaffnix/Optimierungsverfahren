%___________________________________________________________________
\begin{frame}{Ein einfaches Beispiel}
    Blackbox $b$:
    \begin{itemize}
        \item Eingabe: Bild
        \item Ausgabe: Klasse (Hund, Katze, Auto, \dots)
        \item Ergebnis jedoch nicht immer korrekt
    \end{itemize}

    \begin{align*}
        b(\text{Bild}) \;\rightarrow\; \text{Klasse}
    \end{align*}

    Training der Blackbox:
    \begin{itemize}
        \item Anpassen der Parameter von $b$, wenn die falsche Klasse vorhergesagt wurde
        \item Größe: ca. 500\,000 Parameter
        \item Datensatz: 60\,000 Bilder
    \end{itemize}
\end{frame}
